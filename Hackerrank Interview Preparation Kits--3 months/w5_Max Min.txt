def maxMin(k, arr):
    # Write your code here
    # sort
    arr.sort()
    # choose min difference
    mindiff=arr[k]-arr[0]
    for i in range(len(arr)-k+1):
        if mindiff > arr[i+k-1]-arr[i]:
            mindiff = arr[i+k-1]-arr[i]
    return mindiff
