def balancedSums(arr):
    # Write your code here
    presum=[]
    s=0
    for num in arr:
        s += num
        presum.append(s)
    if s-arr[0]==0:
        return "YES"
    for i in range(1,len(arr)):
        if presum[i-1] == s-presum[i]:
            return "YES"
    return "NO"
