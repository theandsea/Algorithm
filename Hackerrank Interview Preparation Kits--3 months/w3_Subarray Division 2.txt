#  1. INTEGER_ARRAY s
#  2. INTEGER d
#  3. INTEGER m
#

def birthday(s, d, m):
    # Write your code here
    if len(s) < m:
        return 0
    t=0
    for i in range(len(s)-m+1):
        if sum(s[i:i+m])==d:
            t += 1
    return t

if __name__ == '__main__':

    result = birthday(s, d, m)
