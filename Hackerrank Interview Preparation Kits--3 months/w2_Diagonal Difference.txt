#

def diagonalDifference(arr):
    # Write your code here
    n=len(arr)
    plus=0
    for i in range(n):
        plus += arr[i][i]
        
    minus=0
    for i in range(n):
        minus += arr[i][n-1-i]
    
    return abs(plus-minus)

if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    n = int(input().strip())

    arr = []

    for _ in range(n):
        arr.append(list(map(int, input().rstrip().split())))

    result = diagonalDifference(arr)
