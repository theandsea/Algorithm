def minimumAbsoluteDifference(arr):
    # Write your code here
    arr.sort()
    mindiff=arr[-1]-arr[0]
    for i in range(len(arr)-1):
        if mindiff > arr[i+1]-arr[i]:
            mindiff = arr[i+1]-arr[i]
    
    return mindiff
